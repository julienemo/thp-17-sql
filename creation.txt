MOOCademy

CREATE TABLE courses(`CourseId` INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL, `Name` NVARCHAR(120) NOT NULL, `Description` TEXT NOT NULL);

CREATE TABLE lessons(
   `LessonId` INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
   `CourseId` INTEGER NOT NULL,
   `Title` NVARCHAR(120),
   `Body` TEXT,
   FOREIGN KEY (CourseId) REFERENCES courses (CourseId)
   ON DELETE CASCADE ON UPDATE NO ACTION
);

Hacking News
CREATE TABLE users(
   `UserId` INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
   `Name` NVARCHAR(120)
);

CREATE TABLE links(
   `LinkId` INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
   `Url` VARCHAR,
   `UserId` INTEGER NOT NULL,
   FOREIGN KEY( UserId ) REFERENCES users (UserId)
   ON DELETE CASCADE ON UPDATE NO ACTION
);

CREATE TABLE comments(
   `CommentId` INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
   `Content` TEXT,
   `UserId` INTEGER NOT NULL,
   `LinkId` INTEGER NOT NULL,
   FOREIGN KEY( UserId ) REFERENCES users (UserId)
   ON DELETE CASCADE ON UPDATE NO ACTION,
   FOREIGN KEY( LinkId ) REFERENCES links (LinkId)
   ON DELETE CASCADE ON UPDATE NO ACTION
);
CREATE TABLE bis_comments(
   `BisCommentId` INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
   `Content` TEXT,
   `UserId` INTEGER NOT NULL,
   `CommentId` INTEGER NOT NULL,
   FOREIGN KEY( UserId ) REFERENCES users (UserId)
   ON DELETE CASCADE ON UPDATE NO ACTION,
   FOREIGN KEY( CommentId ) REFERENCES comments (CommentId)
   ON DELETE CASCADE ON UPDATE NO ACTION
);
